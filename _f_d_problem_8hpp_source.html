407  std::vector<int> nodes[6] = {
408  left_nodes, right_nodes,
409  bottom_nodes, top_nodes,
410  back_nodes, front_nodes
411  };
412  // indices of the in-plane coordinates for each boundaries
413  // 0 for x, 1 for y, 2 for z
414  int ind[6][2] = {{1,2},{1,2},{0,2},{0,2},{0,1},{0,1}};
415  T d;
416  // space increment normal to the boundaries
417  T d_i[6] = {dx, dx, dy, dy, dz, dz};
418  // the number of boundary to set up depend on the dimension
419  int n_bnd = 2 * m_dim;
420  const Coord& c = coordinates();
421  for(int bnd=0; bnd<n_bnd; bnd++){
422  if (m_bc_types[bnd] == 0){ // Dirichlet
423  for(int i=0; i<nodes[bnd].size(); i++){
424  // the rhs value is equal to the value returned by the
425  // boundary function
426  rhs[nodes[bnd][i]] = (this->*m_functions[bnd]) (
427  c[nodes[bnd][i]][ind[bnd][0]],
428  c[nodes[bnd][i]][ind[bnd][1]],
429  t + dt);
430  }
431  } else if(m_bc_types[bnd] == 1){ // Neumann
432  T sign = 1.0, g, g_n, alpha_m, alpha_p;
433  int k, side = 1;
434  for(int i=0; i<nodes[bnd].size(); i++){ // scaled by 1/2
435  k = nodes[bnd][i];
436  g = (this->*m_functions[bnd]) (
437  c[k][ind[bnd][0]],
438  c[k][ind[bnd][1]],
439  t + dt);
440  g_n = (this->*m_functions[bnd]) (
441  c[k][ind[bnd][0]],
442  c[k][ind[bnd][1]],
443  t);
444  if (bnd % 2){ // right, top, front (right side outside)
445  sign = 1.0; // outward --> normal unit vector
446  side = -1;
447  alpha_m = 0.5 * (alpha[k] + alpha[k-1]);
448  // TODO get alpha on the right of the boundary
449  alpha_p = 0.5 * (alpha[k] + alpha[k]);
450  } else{ // left, bottom, back (left side outside)
451  sign = -1.0; // outward --> normal unit vector
452  side = 1;
453  // TODO get alpha on the left of the boundary
454  alpha_m = 0.5 * (alpha[k] + alpha[k]);
455  alpha_p = 0.5 * (alpha[k] + alpha[k+1]);
456  }

464  sign*2.0*d*alpha[k]*g_n*d_i[bnd] + dt*f_n[k])
465  );
